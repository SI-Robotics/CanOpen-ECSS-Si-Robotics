132 int canopen_space_init(const canopen_space_cfg_t* cfg)
133 {
134  if (!cfg) return -1;
135  if (!cfg->send) return -2;
136  if (!cfg->mem || !cfg->mem_size) return -3;
137  memset(&g_ctx, 0, sizeof(g_ctx));
138  g_ctx.role = cfg->role;
139  g_ctx.node_id = cfg->node_id;
140  g_ctx.send = cfg->send;

143  g_ctx.pool_size = cfg->mem_size;
144 
145  alloc_t* alloc = mempool_init(&g_ctx.pool, g_ctx.pool_mem, g_ctx.pool_size);
146  if (!alloc) die(1);
147  g_ctx.net = can_net_create(alloc, 0);
148  if (!g_ctx.net) die(2);

150  can_net_set_next_func(g_ctx.net, &can_next_func, NULL);
151  g_ctx.ts.tv_sec = 0;
152  g_ctx.ts.tv_nsec = 1000000L;
153  (void)can_net_set_time(g_ctx.net, &g_ctx.ts);
154  g_ctx.dev = od_init_for_role(g_ctx.role);
155  if (!g_ctx.dev) die(3);
156  (void)co_dev_set_id(g_ctx.dev, (co_unsigned8_t)g_ctx.node_id);
157  if (g_ctx.role == CO_ROLE_ACTUATOR)
158  {
159  const co_unsigned16_t hb_ms = 100;
160  if (co_dev_set_val(g_ctx.dev, 0x1017, 0x00, &hb_ms, sizeof(hb_ms)) != sizeof(hb_ms))
161  die(34);
162  }
163  else
164  {
165  const co_unsigned8_t n = 1;
166  const co_unsigned32_t hb_cons = ((co_unsigned32_t)300 << 16) | (co_unsigned32_t)2;
167 
168  if (co_dev_set_val(g_ctx.dev, 0x1016, 0x00, &n, sizeof(n)) != sizeof(n))

171  die(41);
172  }
173  set_errc(0);
174  g_ctx.nmt = co_nmt_create(g_ctx.net, g_ctx.dev);
175  if (!g_ctx.nmt) die(6);
176  if (g_ctx.role == CO_ROLE_ACTUATOR)
177  {
178  pdo_config_slave_runtime(g_ctx.dev, g_ctx.node_id);
179 
180  g_ctx.rpdo1 = co_rpdo_create(g_ctx.net, g_ctx.dev, 1);

186  }
187 
188  return 0;
